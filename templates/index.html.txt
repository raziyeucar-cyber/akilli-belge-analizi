    <script>
        async function sorgula() {
            const sorgu = document.getElementById('sorguInput').value;
            const sonucDiv = document.getElementById('sonuc');
            
            if (!sorgu.trim()) {
                alert("Lütfen bir metin girin!");
                return;
            }

            sonucDiv.innerHTML = "Yapay zeka yanıtı hazırlıyor, lütfen bekleyin...";

            try {
                const response = await fetch('/api/chat', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ sorgu: sorgu })
                });

                const data = await response.json();
                if (response.ok) {
                    sonucDiv.innerHTML = "<strong>Sonuç:</strong><br>" + data.cevap;
                } else {
                    sonucDiv.innerHTML = "Hata oluştu: " + (data.hata || "Bilinmeyen hata");
                }
            } catch (error) {
                sonucDiv.innerHTML = "Bağlantı hatası: " + error;
            }
        }
    </script>
